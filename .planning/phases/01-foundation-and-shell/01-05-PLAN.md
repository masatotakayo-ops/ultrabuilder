    Create .github/workflows/ci.yml with the EXACT workflow from RESEARCH.md, with these specifications:

    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      test:
        strategy:
          fail-fast: false
          matrix:
            include:
              - platform: macos-latest
                rust-target: aarch64-apple-darwin
              - platform: macos-latest
                rust-target: x86_64-apple-darwin
              - platform: ubuntu-22.04
                rust-target: x86_64-unknown-linux-gnu
              - platform: windows-latest
                rust-target: x86_64-pc-windows-msvc
        runs-on: ${{ matrix.platform }}

        steps:
          - uses: actions/checkout@v4

          - name: Setup pnpm
            uses: pnpm/action-setup@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: 22
              cache: pnpm

          - name: Install Rust stable
            uses: dtolnay/rust-toolchain@stable
            with:
              targets: ${{ matrix.rust-target }}

          - name: Rust cache
            uses: swatinem/rust-cache@v2
            with:
              workspaces: apps/desktop/src-tauri -> target

          - name: Install Linux dependencies
            if: matrix.platform == 'ubuntu-22.04'
            run: |
              sudo apt-get update
              sudo apt-get install -y \
                libwebkit2gtk-4.1-dev \
                libappindicator3-dev \
                librsvg2-dev \
                patchelf

          - name: Install frontend dependencies
            run: pnpm install --frozen-lockfile

          - name: Lint
            run: pnpm turbo lint

          - name: Test (frontend)
            run: pnpm turbo test

          - name: Build packages
            run: pnpm turbo build --filter=@ultrabuilder/shared

          - name: Build Tauri app
            uses: tauri-apps/tauri-action@v0
            env:
              GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
            with:
              projectPath: apps/desktop
              args: --target ${{ matrix.rust-target }}
              tauriScript: pnpm tauri
    ```

    Key points:
    - 4 matrix targets: macOS ARM (aarch64), macOS Intel (x86_64), Ubuntu 22.04, Windows
    - fail-fast: false so all platforms complete even if one fails
    - pnpm cache via pnpm/action-setup@v4 + actions/setup-node cache
    - Rust cache via swatinem/rust-cache@v2 scoped to src-tauri workspace
    - Linux deps: libwebkit2gtk-4.1-dev (Tauri WebView requirement on Linux)
    - Lint -> Test -> Build packages -> Build Tauri app (sequential within each matrix job)
    - Uses tauri-apps/tauri-action@v0 for the actual Tauri build (handles platform-specific bundling)

    Create the .github/workflows/ directory if it does not exist.
