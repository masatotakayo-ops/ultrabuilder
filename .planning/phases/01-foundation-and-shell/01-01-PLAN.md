       ```json
           "target": "ES2021",
           "module": "ESNext",

           "esModuleInterop": true,
           "skipLibCheck": true,
           "declaration": true,
           "declarationMap": true,
           "outDir": "./dist",
           "rootDir": "./src"
       ```

    3. Create packages/shared/src/schema.ts with the Projects table definition per D-06:
       ```typescript
       import { sqliteTable, text, integer } from "drizzle-orm/sqlite-core";

       export const projects = sqliteTable("projects", {
         id: text("id").primaryKey(),
         name: text("name").notNull(),
         description: text("description").notNull().default(""),
         githubRepo: text("github_repo"),
         status: text("status", { enum: ["planning", "building", "complete"] })
           .notNull()
           .default("planning"),
         createdAt: integer("created_at", { mode: "timestamp" })
           .notNull()
           .$defaultFn(() => new Date()),
         updatedAt: integer("updated_at", { mode: "timestamp" })
           .notNull()
           .$defaultFn(() => new Date()),
       });
       ```

    4. Create packages/shared/src/types.ts:
       ```typescript
       import type { projects } from "./schema";

       export type Project = typeof projects.$inferSelect;
       export type NewProject = typeof projects.$inferInsert;
       ```

    5. Create packages/shared/src/index.ts:
       ```typescript
       export { projects } from "./schema";
       export type { Project, NewProject } from "./types";
       ```

       ```json
         }
       }
       ```
