    **Channel Hook (for log streaming):**

    1. Create apps/desktop/src/hooks/use-tauri-channel.ts:
       ```typescript
       import { useEffect, useRef } from "react";
       import { invoke, Channel } from "@tauri-apps/api/core";

       interface LogEvent {
         event: "logMessage";
         data: {
           level: string;
           target: string;
           message: string;
           timestamp: number;
         };
       }

       export function useLogStream(onEvent: (event: LogEvent) => void) {
         const callbackRef = useRef(onEvent);
         callbackRef.current = onEvent;

         useEffect(() => {
           let active = true;
           const channel = new Channel<LogEvent>();
           channel.onmessage = (event) => {
             if (active) callbackRef.current(event);
           };
           invoke("start_log_stream", { onEvent: channel }).catch(console.error);

           return () => { active = false; };
         }, []);
       }
       ```
       Per RESEARCH.md Pitfall 6: set active=false on cleanup to prevent state updates on unmounted components.

    2. Create apps/desktop/src/hooks/use-tauri-event.ts:
       ```typescript
       import { useEffect } from "react";
       import { listen } from "@tauri-apps/api/event";

       export function useTauriEvent<T = void>(
         eventName: string,
         handler: (payload: T) => void
       ) {
         useEffect(() => {
           let unlisten: (() => void) | undefined;
           listen<T>(eventName, (event) => handler(event.payload))
             .then((fn) => { unlisten = fn; });
           return () => { unlisten?.(); };
         }, [eventName]);
       }
       ```
       Wire up lifecycle events in the root layout or App.tsx:

    **IPC Debug Store:**

    3. Create apps/desktop/src/stores/ipc-store.ts:
       ```typescript
       import { create } from "zustand";

       export interface IpcEvent {
         id: number;
         timestamp: number;
         type: "command" | "channel" | "event";
         direction: "invoke" | "response" | "emit" | "listen";
         name: string;
         payload: unknown;
       }

       interface IpcState {
         events: IpcEvent[];
         nextId: number;
         addEvent: (event: Omit<IpcEvent, "id">) => void;
         clearEvents: () => void;
       }

       const MAX_EVENTS = 500;

       export const useIpcStore = create<IpcState>((set) => ({
         events: [],
         nextId: 1,
         addEvent: (event) =>
           set((s) => ({
             events: [...s.events.slice(-(MAX_EVENTS - 1)), { ...event, id: s.nextId }],
             nextId: s.nextId + 1,
           })),
         clearEvents: () => set({ events: [], nextId: 1 }),
       }));
       ```
       Ring buffer capped at 500 events (per ARCHITECTURE.md Pitfall 8: memory pressure).
